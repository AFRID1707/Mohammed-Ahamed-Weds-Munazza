  <!-- Three.js + Controls (ES modules from a fast CDN) -->
  <script type="module">
    import * as THREE from "https://unpkg.com/three@0.160.1/build/three.module.js";
    import { OrbitControls } from "https://unpkg.com/three@0.160.1/examples/jsm/controls/OrbitControls.js";

    // Basic scene
    const container = document.getElementById("scene");
    const scene = new THREE.Scene();

    // Camera
    const camera = new THREE.PerspectiveCamera(45, container.clientWidth / container.clientHeight, 0.1, 100);
    camera.position.set(0.8, 0.6, 2.2);

    // Renderer
    const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true, powerPreference: "high-performance" });
    renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
    renderer.setSize(container.clientWidth, container.clientHeight);
    renderer.shadowMap.enabled = false;
    container.appendChild(renderer.domElement);

    // Subtle environment feel
    scene.background = null;
    const hemi = new THREE.HemisphereLight(0xffffff, 0x202030, 0.65);
    hemi.position.set(0, 1, 0);
    scene.add(hemi);

    // Gold-ish highlight that moves as card rotates
    const keyLight = new THREE.DirectionalLight(0xffe7a3, 0.65);
    keyLight.position.set(2.0, 1.5, 1.8);
    scene.add(keyLight);

    const rimLight = new THREE.DirectionalLight(0xffffff, 0.2);
    rimLight.position.set(-2.5, -1.0, -2.0);
    scene.add(rimLight);

    // Card geometry (w:h ~ 2:3 or 3:4; adjust to match your artwork ratio)
    // Using 3:4 portrait (0.9w x 1.2h meters). Increase segments for smoother anisotropy.
    const CARD_WIDTH = 0.90;
    const CARD_HEIGHT = 1.20;
    const CARD_THICKNESS = 0.006; // 6mm visual thickness

    // Texture loader with nice filtering
    const loader = new THREE.TextureLoader();
    function loadTex(path) {
      const t = loader.load(path);
      t.colorSpace = THREE.SRGBColorSpace;
      t.anisotropy = Math.min(8, renderer.capabilities.getMaxAnisotropy());
      t.minFilter = THREE.LinearMipmapLinearFilter;
      t.magFilter = THREE.LinearFilter;
      return t;
    }

    const frontTex = loadTex("./front.jpg");
    const backTex  = loadTex("./back.jpg");

    // Materials: slightly glossy to give a foil-like shimmer to gold areas
    const frontMat = new THREE.MeshPhysicalMaterial({
      map: frontTex,
      metalness: 0.1,
      roughness: 0.45,
      reflectivity: 0.35,
      clearcoat: 0.15,
      clearcoatRoughness: 0.6
    });

    const backMat = new THREE.MeshPhysicalMaterial({
      map: backTex,
      metalness: 0.08,
      roughness: 0.5,
      reflectivity: 0.3,
      clearcoat: 0.12,
      clearcoatRoughness: 0.65
    });

    // We’ll create a thin box for realistic edges, then assign different materials
    const cardGeo = new THREE.BoxGeometry(CARD_WIDTH, CARD_HEIGHT, CARD_THICKNESS, 1, 1, 1);

    // BoxGeometry has 6 faces; we want:
    //  - front face (positive Z) -> frontMat
    //  - back face (negative Z) -> backMat
    //  - edges -> a dark navy material
    const edgeMat = new THREE.MeshPhysicalMaterial({
      color: new THREE.Color("#0b1020"),
      metalness: 0.05,
      roughness: 0.65
    });

    // Material order for BoxGeometry: +X, -X, +Y, -Y, +Z, -Z
    const materials = [edgeMat, edgeMat, edgeMat, edgeMat, frontMat, backMat];
    const card = new THREE.Mesh(cardGeo, materials);
    card.rotation.x = THREE.MathUtils.degToRad(-8);
    scene.add(card);

    // Soft shadow ground (faux) for depth
    const groundGeo = new THREE.PlaneGeometry(5, 5);
    const groundMat = new THREE.MeshBasicMaterial({
      color: 0x000000,
      transparent: true,
      opacity: 0.15
    });
    const ground = new THREE.Mesh(groundGeo, groundMat);
    ground.rotation.x = -Math.PI / 2;
    ground.position.y = -0.75;
    ground.renderOrder = -1;
    scene.add(ground);

    // Subtle card float animation
    let t0 = performance.now();
    function floatCard(time) {
      const t = (time - t0) * 0.001;
      card.position.y = Math.sin(t * 0.6) * 0.015; // gentle bob
      keyLight.position.x = 2.0 + Math.sin(t * 0.5) * 0.2; // tiny light drift for shimmer
    }

    // Controls
    const controls = new OrbitControls(camera, renderer.domElement);
    controls.enableDamping = true;
    controls.dampingFactor = 0.08;
    controls.enablePan = false;
    controls.minDistance = 1.2;
    controls.maxDistance = 4.0;
    controls.minPolarAngle = THREE.MathUtils.degToRad(35);
    controls.maxPolarAngle = THREE.MathUtils.degToRad(85);
    controls.target.set(0, 0, 0);

    // Double-click/tap to reset view
    const resetView = () => {
      controls.target.set(0, 0, 0);
      camera.position.set(0.8, 0.6, 2.2);
      controls.update();
    };
    renderer.domElement.addEventListener("dblclick", resetView, { passive: true });
    renderer.domElement.addEventListener("touchend", (e) => {
      if (e.touches && e.touches.length === 0 && e.changedTouches && e.changedTouches.length === 1) {
        // Simple double-tap detector
        const now = Date.now();
        if (!window.__lastTapTime) window.__lastTapTime = 0;
        if (now - window.__lastTapTime < 300) resetView();
        window.__lastTapTime = now;
      }
    }, { passive: true });

    // Handle resize
    function onResize() {
      const w = container.clientWidth;
      const h = container.clientHeight;
      renderer.setSize(w, h);
      camera.aspect = w / h;
      camera.updateProjectionMatrix();
    }
    window.addEventListener("resize", onResize);
    onResize();

    // Render loop
    function animate(time) {
      requestAnimationFrame(animate);
      floatCard(time || 0);
      controls.update();
      renderer.render(scene, camera);
    }
    animate();

    // Helpful console tip for file names
    console.log("Make sure your images are named exactly front.jpg and back.jpg and placed next to index.html.");
  </script>
